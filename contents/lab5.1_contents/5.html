Напишіть функцію конструктор Storage, який створює об'єкти для управління складом товарів. При виклику отримуватиме один агрумент - початковий масив товарів, і записувати їх у властивість items. Додайте методи класу:
getItems() - повертайте масив товарів
addItems(item) - отримує новий товар та додає його до поточних
removeItem(item) - отримує товар і, якщо він є, видаляє його з поточних

function Storage(items) {
    this.items = items;

    this.getItems = function() {
        return this.items;
    };

    this.addItem = function(item) {
        this.items.push(item);
    };

    this.removeItem = function(item) {
        const index = this.items.indexOf(item);
        if (index > -1) {
            this.items.splice(index, 1);
        }
        else {
            console.log("Такого товару не існує в списку.");
        }
    };
}
