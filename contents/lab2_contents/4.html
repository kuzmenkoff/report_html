Селектори класу

Налаштування меню навігації з використанням селекторів класу:

    
HTML:
<nav>
    <ul class="navigation">
        <li><a style="text-decoration: underline; text-underline-offset: 5px;" href="index.html">Home</a></li>
        <li><a href="catalog.html">Catalog</a></li>
    </ul>
</nav>

CSS:
.navigation {
    display: flex;
    list-style-type: none;
    margin: 0;
    padding: 0;
    overflow: hidden;

}

.navigation li {
    float: left;
    margin-right: 45px;
    margin-left: 45px;
}

.navigation a {
    color: black;
    font-size: 18px;
    font-weight: bold; 
    display: block;
    text-align: center;
    padding: 14px 16px;
    text-decoration: none;
}

.navigation a:hover {
    color: white;
}
